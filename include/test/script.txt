
local key = 0;

local snake = table();

snake.head = table();
snake.tails = array();
snake.lasttail = 1;
snake.lastpos = table();

snake.head.x = 0;
snake.head.y = 0;
snake.head.shape = "x";

local food = table();
food.x = random()%70;
food.y = random()%25;
food.shape = "f";

while (1)
{
label continue:
	key = getch();

	snake.lastpos.x = snake.head.x;
	snake.lastpos.y = snake.head.y;

	if ( key == 'w' ) --snake.head.y;
	else if ( key == 's' ) ++snake.head.y;
	else if ( key == 'd' ) ++snake.head.x;
	else if ( key == 'a' ) --snake.head.x;
	else if ( key == 'x' ) break;
	else goto continue;

	local j = snake.tails:size();
	while (j)
	{
		local tail = snake.tails[j];
		if ( (tail.x==snake.head.x) && (tail.y==snake.head.y) ) 
		{
			cls();
			gotoxy( 35, 13 );
			print("game over\n");
			getch();
			return;
		}
		--j;
	}

	if ( (food.x==snake.head.x) && (food.y==snake.head.y) )
	{
		local tail = table();
		tail.x = snake.lastpos.x;
		tail.y = snake.lastpos.y;
		tail.shape = "0";
		snake.tails:push(tail);
		food.x = random()%70;
		food.y = random()%25;
	}
	else if ( snake.tails:size() )
	{
		local tail = snake.tails[snake.lasttail];
		tail.x = snake.lastpos.x;
		tail.y = snake.lastpos.y;
		++snake.lasttail;
		if ( snake.lasttail > snake.tails:size() ) snake.lasttail = 1;
	}

	cls();
	gotoxy( 0, 0 );
	print( snake.tails:size() );
	gotoxy( snake.head.x, snake.head.y );
	print( snake.head.shape );
	gotoxy( food.x, food.y );
	print( food.shape );
	
	local i = snake.tails:size();
	while (i)
	{
		local tail = snake.tails[i];
		gotoxy( tail.x, tail.y );
		print( tail.shape );
		--i;
	}

}
