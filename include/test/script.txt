
local meta	= table();
local t		= table();

meta.__call = function() { print("call"); };
meta.__add = function(a,b) { print("add : "+b); };
meta.__sub = function(a,b) { print("sub : "+b); };
meta.__mul = function(a,b) { print("mul : "+b); };
meta.__div = function(a,b) { print("div : "+b); };
meta.__mod = function(a,b) { print("mod : "+b); };

meta.__eq = function(a,b) { print("__eq : "+b); };
meta.__neq = function(a,b) { print("__neq : "+b); };
meta.__lt = function(a,b) { print("__lt : "+b); };
meta.__gt = function(a,b) { print("__gt : "+b); };

setmeta( t, meta );

t();
print( t+10 );
print( t-15 );
print( t*44 );
print( t/76 );
print( t%5 );

print( t==15 );
print( t!=44 );
print( t<76 );
print( t>5 );
